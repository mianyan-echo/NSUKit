83 template<class T>
84 T JsonWrapper::get_icd_param(nsuCSParam_t &param_name) {
85 Json::Value reg, value;
86 T _value;
87 if (!icdParams->isMember(param_name)) throw std::runtime_error("此参数不存在");
88 reg = (*icdParams)[param_name];
89 value = reg[1];
90
91 if (value.isString() && StringStartWith(value.asString(), "0x")) {
92 _value = (T) std::stoll(value.asString(), nullptr, 16);
93 } else if (value.isString() && StringStartWith(value.asString(), "0b")) {
94 _value = (T) std::stoll(value.asString().substr(2), nullptr, 2);
95 } else if (value.isString() && reg[0].asString() == ICD_File && std::is_base_of<T, std::string>::value) {
96 _value = get_icd_param<T>(param_name);
97 } else if (value.isNumeric()) {
98 std::string value_type = reg[0].asString();
99 if (value_type == ICD_Float) {
100 _value = (T) value.asFloat();
101 } else if (value_type == ICD_Double) {
102 _value = (T) value.asDouble();
103 } else if (value_type == ICD_Uint8) {
104 _value = (T) value.asUInt();
105 } else if (value_type == ICD_Int8) {
106 _value = (T) value.asInt();
107 } else if (value_type == ICD_Uint16) {
108 _value = (T) value.asUInt();
109 } else if (value_type == ICD_Int16) {
110 _value = (T) value.asInt();
111 } else if (value_type == ICD_Uint32) {
112 _value = (T) value.asUInt();
113 } else if (value_type == ICD_Int32) {
114 _value = (T) value.asInt();
115 } else {
116 std::cout << "未识别的值类型 " << value.toStyledString() << std::endl;
117 throw std::runtime_error("未识别的寄存器值类型");
118 }
119 } else {
120 std::cout << "未识别的值类型 " << value.toStyledString() << std::endl;
121 throw std::runtime_error("未识别的寄存器值类型");
122 }
123 return _value;
124 }
