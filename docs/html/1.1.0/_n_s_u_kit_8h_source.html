179 NSUSoc<CSItf_t, CRItf_t, DSItf_t, CmdMw_t, ChnlMw_t>::NSUSoc() {
180 itf_cs_typesafe = std::is_base_of<I_BaseCmdUItf, CSItf_t>::value;
181 itf_cr_typesafe = std::is_base_of<I_BaseCmdUItf, CRItf_t>::value;
182 itf_ds_typesafe = std::is_base_of<I_BaseStreamUItf, DSItf_t>::value;
183 mw_cmd_typesafe = std::is_base_of<I_BaseRegMw, CmdMw_t>::value;
184 mw_stream_typesafe = std::is_base_of<I_BaseStreamMw, ChnlMw_t>::value;
185
186 itf_cs = new CSItf_t();
187 if (std::is_same<CSItf_t, CRItf_t>::value) {
188 itf_cr = (CRItf_t *)itf_cs;
189 } else {
190 itf_cr = new CRItf_t();
191 }
192 itf_ds = new DSItf_t();
193 mw_cmd = new CmdMw_t(itf_cs);
194 mw_chnl = new ChnlMw_t(itf_ds);
195 }

227 template<class CSItf_t, class CRItf_t, class DSItf_t, class CmdMw_t, class ChnlMw_t>
228 nsukitStatus_t NSUSoc<CSItf_t, CRItf_t, DSItf_t, CmdMw_t, ChnlMw_t>::link_cmd(nsuInitParam_t *param) {
229 if (param == nullptr) {
230 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
231 }
232 METHOD_NEED_(check_typesafe);
233
234 // 初始化状态为成功
235 nsukitStatus_t status = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
236
237 if (!this->combined_cmd_itf()) {
238 auto* crInterface = dynamic_cast<I_BaseCmdUItf*>(itf_cr);
239 status |= crInterface->accept(param);
240 }
241 // 调用命令接口的 accept 函数，将结果合并到状态中
242 auto* csInterface = dynamic_cast<I_BaseCmdUItf*>(itf_cs);
243 status |= csInterface->accept(param);
244 // 调用命令中间件的 config 函数，将结果合并到状态中
245 auto* cmdMiddleware = dynamic_cast<I_BaseRegMw*>(mw_cmd);
246 status |= cmdMiddleware->config(param);
247
248 // 返回合并后的状态
249 return status;
250 }
