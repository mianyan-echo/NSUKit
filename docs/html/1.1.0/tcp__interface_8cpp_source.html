15nsukitStatus_t TCPCmdUItf::accept(nsuInitParam_t *param) {
16 tcpTimeout = param->cmd_tcp_timeout;
17 tcpPort = param->cmd_tcp_port;
18 ipAddr = param->cmd_ip;
19
20 opLock.lock();
21 try {
22 sockGen.Init(true, tcpPort, ipAddr);
23 sockGen.ConnectServer();
24 } catch (...) {
25 opLock.unlock();
26 return nsukitStatus_t::NSUKIT_STATUS_ACCEPT_FAIL;
27 }
28 opLock.unlock();
29 return nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
30}

59nsukitStatus_t TCPCmdUItf::send_bytes(nsuCharBuf_p bytes, nsuSize_t length) {
60 auto res = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
61 opLock.lock();
62 auto tcp_status = TcpSendBytes(&sockGen, bytes, length);
63 if (tcp_status == 0) res |= nsukitStatus_t::NSUKIT_STATUS_ITF_FAIL;
64
65// sockGen.CloseSock(sockGen.tcpClient);
66 opLock.unlock();
67 return res;
68}
