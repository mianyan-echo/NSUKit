175 NSUSoc<CSItf_t, CRItf_t, DSItf_t, CmdMw_t, ChnlMw_t>::NSUSoc() {
176 itf_cs_typesafe = std::is_base_of<I_BaseCmdUItf, CSItf_t>::value;
177 itf_cr_typesafe = std::is_base_of<I_BaseCmdUItf, CRItf_t>::value;
178 itf_ds_typesafe = std::is_base_of<I_BaseStreamUItf, DSItf_t>::value;
179 mw_cmd_typesafe = std::is_base_of<I_BaseRegMw, CmdMw_t>::value;
180 mw_stream_typesafe = std::is_base_of<I_BaseStreamMw, ChnlMw_t>::value;
181
182 itf_cs = new CSItf_t();
183 if (std::is_same<CSItf_t, CRItf_t>::value) {
184 itf_cr = (CRItf_t *)itf_cs;
185 } else {
186 itf_cr = new CRItf_t();
187 }
188 itf_ds = new DSItf_t();
189 mw_cmd = new CmdMw_t(itf_cs);
190 mw_chnl = new ChnlMw_t(itf_ds);
191 }

223 template<class CSItf_t, class CRItf_t, class DSItf_t, class CmdMw_t, class ChnlMw_t>
224 nsukitStatus_t NSUSoc<CSItf_t, CRItf_t, DSItf_t, CmdMw_t, ChnlMw_t>::link_cmd(nsuInitParam_t *param) {
225 if (param == nullptr) {
226 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
227 }
228 METHOD_NEED_(check_typesafe);
229
230 // 初始化状态为成功
231 nsukitStatus_t status = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
232
233 if (!this->combined_cmd_itf()) {
234 auto* crInterface = dynamic_cast<I_BaseCmdUItf*>(itf_cr);
235 status |= crInterface->accept(param);
236 }
237 // 调用命令接口的 accept 函数，将结果合并到状态中
238 auto* csInterface = dynamic_cast<I_BaseCmdUItf*>(itf_cs);
239 status |= csInterface->accept(param);
240 // 调用命令中间件的 config 函数，将结果合并到状态中
241 auto* cmdMiddleware = dynamic_cast<I_BaseRegMw*>(mw_cmd);
242 status |= cmdMiddleware->config(param);
243
244 // 返回合并后的状态
245 return status;
246 }
