4#include <iostream>
5#include <fstream>
6#include "NSUKit.h"
7
8
9int main(int argc, char *argv[]) {
10 const int ds_chnl = 0;
11 std::ofstream outf;
12 nsukit::NSUSoc<nsukit::TCPCmdUItf, nsukit::PCIECmdUItf, nsukit::PCIEStreamUItf> kit{};
13 nsuInitParam_t param;
14
15 std::cout << "当前示例" << argv[0]
16 << "为展示基于PCIE的数据上行的接口调用示例，功能为从板卡0的数传通道0上行指定数量的数据并写入文件" << std::endl;
17 if (argc != 4) {
18 std::cout << "不受支持的传参方法" << std::endl;
19 std::cout << argv[0] << " {IP} {totalBytes} {filePath}" << std::endl;
20 return 1;
21 }
22 nsuSize_t total_len = std::atoi(argv[2]);
23 if (total_len % 4 != 0) {
24 std::cout << "上行数据总长度total_len " << total_len << "Bytes应为 " << 4 << "Bytes的整倍数" << std::endl;
25 return 1;
26 }
27
28 // 连接板卡
29 param.cmd_ip = argv[1];
30 param.cmd_board = 0;
31 param.stream_board = 0;
32 auto res = kit.link_cmd(&param);
33 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
34 std::cout << "建立CS、CR连接：" << nsukit::status2_string(res) << std::endl;
35 }
36 res = kit.link_stream(&param);
37 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
38 std::cout << "建立DS连接：" << nsukit::status2_string(res) << std::endl;
39 }
40
41 // 准备DS连接要用的内存
42 nsuMemory_p mem = kit.alloc_buffer(total_len); // 申请内存
43 auto ds_buf = (char *)kit.get_buffer(mem, total_len); // 获取内存首指针
44
45 // 通知FPGA开始采集
46 res = kit.execute("系统开启");
47 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
48 std::cout << "系统开启：" << nsukit::status2_string(res) << std::endl;
49 }
50
51 // 上行一个颗粒度的数据
52 res = kit.open_recv(ds_chnl, mem, total_len, 0);
53 if (res == nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
54 std::cout << "成功开启DS交互" << std::endl;
55 auto ds_state = nsukitStatus_t::NSUKIT_STATUS_STREAM_RUNNING;
56 while (ds_state != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
57 ds_state = kit.wait_stream(mem, 1000.);
58// std::cout << "wait_stream：" << nsukit::status2_string(ds_state) << std::endl;
59 }
60 }
61
62 std::cout << *(uint32_t *)ds_buf << std::endl;
63
64 // 通知FPGA开始采集
65 res = kit.execute("系统停止");
66 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
67 std::cout << "系统停止：" << nsukit::status2_string(res) << std::endl;
68 }
69
70 // 存盘
71 outf.open(argv[3], std::ofstream::binary);
72 outf.write(ds_buf, total_len);
73 outf.close();
74
75 kit.free_buffer(mem); // 释放内存
76 return 0;
77}
