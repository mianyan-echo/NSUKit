45Mixin_NativeRegCmd::increment_write(nsuRegAddr_t addr, nsuVoidBuf_p value, nsuSize_t length, nsuSize_t reg_len) {
46 auto res = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
47 auto num = length / reg_len;
48 auto ptr = (nsuRegValue_t *) value;
49 for (int i=0;i<num;i++){
50 res |= cmd_itf_->write(addr+reg_len*i, *(ptr+i));
51 }
52 if ((length - num*reg_len) > 0) {
53 nsuRegValue_t last = 0;
54 memcpy(&last, ptr+num, length - num*reg_len);
55 res |= cmd_itf_->write(addr+reg_len*num, last);
56 }
57 return res;
58}

147nsukitStatus_t Mixin_VirtualRegCmd::_common_write(nsuRegAddr_t reg, nsuRegValue_t value) {
148 auto buf = (nsuCharBuf_p) malloc(sizeof(RegPack));
149 auto len = _fmt_reg_write(reg, value, buf);
150 auto res = cmd_itf_->send_bytes(buf, len);
151 free(buf);
152
153 char r_head[16];
154 RegPack pack;
155 res |= cmd_itf_->recv_bytes(16, r_head);
156 memcpy(&pack, r_head, 16);
157 res |= cmd_itf_->recv_bytes(pack.length-16, r_head);
158 if (*(nsuRegValue_t *)r_head!=0) res |= nsukitStatus_t::NSUKIT_STATUS_ITF_FAIL;
159 return res;
160}
