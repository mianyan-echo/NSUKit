256 NSUSoc<CSItf_t, CRItf_t, DSItf_t, CmdMw_t, ChnlMw_t>::NSUSoc() {
257 itf_cs_typesafe = std::is_base_of<I_BaseCmdUItf, CSItf_t>::value;
258 itf_cr_typesafe = std::is_base_of<I_BaseCmdUItf, CRItf_t>::value;
259 itf_ds_typesafe = std::is_base_of<I_BaseStreamUItf, DSItf_t>::value;
260 mw_cmd_typesafe = std::is_base_of<I_BaseRegMw, CmdMw_t>::value;
261 mw_stream_typesafe = std::is_base_of<I_BaseStreamMw, ChnlMw_t>::value;
262
263 itf_cs = new CSItf_t();
264 if (std::is_same<CSItf_t, CRItf_t>::value) {
265 itf_cr = (CRItf_t *)itf_cs;
266 } else {
267 itf_cr = new CRItf_t();
268 }
269 itf_ds = new DSItf_t();
270 mw_cmd = new CmdMw_t(itf_cs);
271 mw_chnl = new ChnlMw_t(itf_ds);
272 }

289 if (param == nullptr) {
290 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
291 }
292 METHOD_NEED_(check_typesafe);
293
294 //
295 nsukitStatus_t status = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
296
297 if (!this->combined_cmd_itf()) {
298 auto* crInterface = dynamic_cast<I_BaseCmdUItf*>(itf_cr);
299 status |= crInterface->accept(param);
300 }
301 //
302 auto* csInterface = dynamic_cast<I_BaseCmdUItf*>(itf_cs);
303 status |= csInterface->accept(param);
304 //
305 auto* cmdMiddleware = dynamic_cast<I_BaseRegMw*>(mw_cmd);
306 status |= cmdMiddleware->config(param);
307
308 if (status == nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
309 cmd_linked = true;
310 }
311
312 //
313 return status;
314 }
