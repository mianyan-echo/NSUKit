158 CommandPack *ICDRegMw::FmtRegister(const Json::Value &reg, const T &value) {
159 auto *res = new CommandPack;
160 int size;
161 size = TypeSizeMap[reg[1].asString()];
162 res->bytes = new char[size + 1]();
163 res->length = size;
164 memcpy(res->bytes, &value, size);
165 return res;
166 }

194 template<class T>
195 T ICDRegMw::GetParamValue(const std::string &param_name) {
196 Json::Value reg, value;
197 T _value;
198 if (!icdParams->isMember(param_name)) throw std::runtime_error("此参数不存在");
199 reg = (*icdParams)[param_name];
200 value = reg[2];
201
202 if (value.isString() && StringStartWith(value.asString(), "0x")) {
203 _value = (T) std::stoll(value.asString(), nullptr, 16);
204 } else if (value.isString() && StringStartWith(value.asString(), "0b")) {
205 _value = (T) std::stoll(value.asString().substr(2), nullptr, 2);
206 } else if (value.isString() && reg[1].asString() == ICD_File) {
207 _value = (T) value.asString();
208 } else if (value.isNumeric()) {
209 std::string value_type = reg[1].asString();
210 if (value_type == ICD_Float) {
211 _value = (T) value.asFloat();
212 } else if (value_type == ICD_Double) {
213 _value = (T) value.asDouble();
214 } else if (value_type == ICD_Uint8) {
215 _value = (T) value.asUInt();
216 } else if (value_type == ICD_Int8) {
217 _value = (T) value.asInt();
218 } else if (value_type == ICD_Uint16) {
219 _value = (T) value.asUInt();
220 } else if (value_type == ICD_Int16) {
221 _value = (T) value.asInt();
222 } else if (value_type == ICD_Uint32) {
223 _value = (T) value.asUInt();
224 } else if (value_type == ICD_Int32) {
225 _value = (T) value.asInt();
226 } else {
227 std::cout << "未识别的值类型 " << value.toStyledString() << std::endl;
228 throw std::runtime_error("未识别的寄存器值类型");
229 }
230 } else {
231 std::cout << "未识别的值类型 " << value.toStyledString() << std::endl;
232 throw std::runtime_error("未识别的寄存器值类型");
233 }
234 return _value;
235 }
