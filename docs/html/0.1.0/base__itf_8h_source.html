267 template<typename T>
268 nsukitStatus_t Mixin_VirtualRegCmd::_fmt_reg_write(nsuRegAddr_t reg, T value, nsuCharBuf_p buf) {
269// static_assert(std::is_integral<T>::value, "T must be an integral type");
270 static constexpr uint32_t DataPacketID = 0x31000000;
271 static constexpr size_t DataPacketLength = sizeof(RegPack) + sizeof(nsuRegAddr_t) + sizeof(T);
272
273 if (buf == nullptr) {
274 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE; // 缓冲区为空，返回错误
275 }
276
277 RegPack pack;
278 pack.id = DataPacketID;
279 pack.length = DataPacketLength;
280
281 memcpy(buf, &pack, sizeof(RegPack));
282 memcpy(buf + sizeof(RegPack), &reg, sizeof(nsuRegAddr_t));
283 memcpy(buf + sizeof(RegPack) + sizeof(nsuRegAddr_t), &value, sizeof(T));
284
285 return nsukitStatus_t::NSUKIT_STATUS_SUCCESS; // 数据格式化成功
286 }

296 template<typename T>
297 nsuBytes_t Mixin_VirtualRegCmd::_fmt_reg_write(nsuRegAddr_t reg, T value) {
298 nsuBytes_t res;
299 uint32_t size = sizeof(RegPack) + sizeof(nsuRegAddr_t) + sizeof(T);
300 char _res[size];
301 RegPack pack;
302 pack.id = 0x31000000;
303 pack.length = size;
304 memcpy(_res, &pack, sizeof(RegPack));
305 memcpy(_res + sizeof(RegPack), &reg, sizeof(nsuRegAddr_t));
306 memcpy(_res + sizeof(RegPack) + sizeof(nsuRegAddr_t), &value, sizeof(T));
307
308 for (auto ch: _res) {
309 res.push_back(ch);
310 }
311 return res;
312 }
