111CommandPack *ICDRegMw::FmtRegister(const Json::Value &reg, const Json::Value &value) {
112 if (value.isString() && StringStartWith(value.asString(), "0x")) {
113 // 对于 "0x5F5F5F5F"
114 auto _value = (uint32_t) std::stoll(value.asString(), nullptr, 16);
115 return FmtRegister(reg, _value);
116 } else if (value.isString() && StringStartWith(value.asString(), "0b")) {
117 // 对于 "0b10101011"
118 auto _value = std::stoll(value.asString().substr(2), nullptr, 2);
119 return FmtRegister(reg, _value);
120 } else if (value.isString() && reg[1].asString() == ICD_File) {
121 // 对于 file文件类型
122 return FmtFile(value.asString());
123 } else if (value.isNumeric()) {
124 // 对于全体数字
125 std::string value_type = reg[1].asString();
126 if (value_type == ICD_Float) {
127 auto _value = value.asFloat();
128 return FmtRegister(reg, _value);
129 } else if (value_type == ICD_Double) {
130 auto _value = value.asDouble();
131 return FmtRegister(reg, _value);
132 } else if (value_type == ICD_Uint8) {
133 auto _value = (uint8_t) value.asUInt();
134 return FmtRegister(reg, _value);
135 } else if (value_type == ICD_Int8) {
136 auto _value = (int8_t) value.asInt();
137 return FmtRegister(reg, _value);
138 } else if (value_type == ICD_Uint16) {
139 auto _value = (uint16_t) value.asUInt();
140 return FmtRegister(reg, _value);
141 } else if (value_type == ICD_Int16) {
142 auto _value = (int16_t) value.asInt();
143 return FmtRegister(reg, _value);
144 } else if (value_type == ICD_Uint32) {
145 auto _value = (uint32_t) value.asUInt();
146 return FmtRegister(reg, _value);
147 } else if (value_type == ICD_Int32) {
148 auto _value = (int32_t) value.asInt();
149 return FmtRegister(reg, _value);
150 } else {
151 std::cout << "未识别的值类型 " << value.toStyledString() << std::endl;
152 throw "未识别的寄存器值类型";
153 }
154 } else {
155 std::cout << "未识别的值类型 " << value.toStyledString() << std::endl;
156 throw "未识别的寄存器值类型";
157 }

186CommandPack *ICDRegMw::FmtFile(const std::string &file_path, bool need_size) {
187 CommandPack *fileData = nullptr;
188 std::ifstream _in(file_path, std::ios::in | std::ios::binary | std::ios::ate);
189 if (_in.is_open()) {
190 fileData = new CommandPack;
191 if (need_size) {
192 fileData->length = TypeSizeMap[ICD_Uint32];
193 fileData->bytes = new char[fileData->length + 1]();
194 uint32_t fileLength = (uint32_t) _in.tellg();
195 memcpy(fileData->bytes, &fileLength, fileData->length);
196 } else {
197 fileData->length = _in.tellg();
198 fileData->bytes = new char[fileData->length + 1]();
199 _in.seekg(0, std::ios::beg);
200 _in.read(fileData->bytes, fileData->length);
201 }
202 }
203 return fileData;
204}

213CommandPack *ICDRegMw::FmtCommand(const std::string &cmd_name, const std::string &file_path) {
214 Json::Value command;
215 std::vector<CommandPack *> registerList;
216 CommandPack *regTemp;
217 auto resCommandPack = new CommandPack;
218 uint32_t commandLength = 0;
219
220 if (icdCommands->isMember(cmd_name)) {
221 command = (*icdCommands)[cmd_name];
222 for (auto &reg: command) {
223 if (reg.isString() && reg.asString() == ICD_ParamFlag_File) {
224 regTemp = FmtFile(file_path);
225 } else if (reg.isString() && reg.asString() == ICD_ParamFlag_FileLength) {
226 regTemp = FmtFile(file_path, true);
227 } else if (reg.isString() && icdParams->isMember(reg.asString())) {
228 reg = (*icdParams)[reg.asString()];
229 regTemp = FmtRegister(reg, reg[2]);
230 } else if (reg.isArray()) {
231 regTemp = FmtRegister(reg, reg[2]);
232 } else {
233 throw "不受支持的寄存器写法";
234 }
235 assert(regTemp->bytes != nullptr);
236 registerList.emplace_back(regTemp);
237 commandLength += regTemp->length;
238 }
239 } else {
240 std::cout << "Not find command: " << cmd_name << std::endl;
241 throw "指令名称在icd中不存在";
242 }
243
244 // 将获取到的二进制寄存器列表拼接为一个CommandPack
245 resCommandPack->bytes = new char[commandLength + 1];
246 resCommandPack->length = commandLength;
247 char *flag = resCommandPack->bytes;
248 for (auto &reg: registerList) {
249 memcpy(flag, reg->bytes, reg->length);
250 flag += reg->length;
251 DeleteCommandPack(reg);
252 }
253 // 修改指令长度
254 memcpy(resCommandPack->bytes + 12, &commandLength, 4);
255
256 return resCommandPack;
257}
