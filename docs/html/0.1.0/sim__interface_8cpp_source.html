107nsukitStatus_t SimStreamUItf::free_buffer(nsuMemory_p fd) {
108 if (fd == nullptr) {
109 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
110 }
111 auto memory = fdMap[fd];
112 fdMap.erase(fd);
113 if (memory->self_malloc) {
114 free(memory->buffer);
115 }
116 free(memory);
117 return nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
