12nsukitStatus_t XDMAOperation_t::open_xdma_board(int board) {
13 OpLock.lock();
14 if (openedBoard.find(board) == openedBoard.end()) {
15 openedBoard[board] = 0;
16 }
17 if (openedBoard[board] == 0) {
18 if (!fpga_open(board)) {
19 OpLock.unlock();
20 return nsukitStatus_t::NSUKIT_STATUS_ACCEPT_FAIL;
21 }
22 }
23 openedBoard[board] += 1;
24 OpLock.unlock();
25 return nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
26}

29nsukitStatus_t XDMAOperation_t::close_xdma_board(int board) {
30 OpLock.lock();
31 if (openedBoard.find(board) == openedBoard.end()) {
32 OpLock.unlock();
33 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
34 }
35 auto num = openedBoard[board];
36 num -= 1;
37 if (num == 0) fpga_close(board);
38 openedBoard[board] = num;
39 OpLock.unlock();
40 return nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
41}

134PCIEStreamUItf::open_send(nsuChnlNum_t chnl, nsuMemory_p fd, nsuStreamLen_t length, nsuStreamLen_t offset) {
135 if ((chnl >= maxChnl) or(length % byteWidth != 0) or (offset % byteWidth != 0)) {
136 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
137 }
138 if (downloadProcess.find(fd) == downloadProcess.end()) {
139 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
140 }
141 auto res = fpga_send(pciBoard, chnl, fd, length/byteWidth, offset/byteWidth, 1, 0, 0, DMA_NOWAIT);
142 if (res == DMA_START_ERROR) {
143 return nsukitStatus_t::NSUKIT_STATUS_STREAM_FAIL;
144 }
145 downloadProcess[fd].total = length;
146 downloadProcess[fd].current = res*byteWidth;
147 uploadProcess[fd].total = 0;
148 return nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
149}

169nsukitStatus_t PCIEStreamUItf::wait_stream(nsuMemory_p fd, float timeout) {
170 StreamProcess *process = nullptr;
171 if (uploadProcess[fd].total != 0) process = &uploadProcess[fd];
172 if (downloadProcess[fd].total != 0) process = &downloadProcess[fd];
173 if (process == nullptr) {
174 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
175 }
176 auto res = fpga_wait_dma(fd, (uint32_t)(timeout*1000.));
177 process->current = res;
178 if (process->current != process->total) {
179 return nsukitStatus_t::NSUKIT_STATUS_STREAM_RUNNING;
180 }
181 return nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
182}

184nsukitStatus_t PCIEStreamUItf::break_stream(nsuMemory_p fd) {

186 if (uploadProcess[fd].total != 0) process = &uploadProcess[fd];
187 if (downloadProcess[fd].total != 0) process = &downloadProcess[fd];
188 if (process == nullptr) {
189 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
190 }
191 fpga_break_dma(fd);
192 process->total = 0;
193 return nsukitStatus_t::NSUKIT_STATUS_SUCCESS;

197PCIEStreamUItf::stream_recv(nsuChnlNum_t chnl, nsuMemory_p fd, nsuStreamLen_t length, nsuStreamLen_t offset,
198 bool(*stop_event) (), float timeout, int flag) {
199 open_recv(chnl, fd, length, offset);
200 if (stop_event == nullptr) {
201 stop_event = []() { return false; };
202 }
203 while (true) {
204 if (stop_event()) {
205 break_stream(fd);
206 return nsukitStatus_t::NSUKIT_STATUS_STREAM_FAIL;
207 }
208
209 auto res = wait_stream(fd, timeout);
210
211 if (res == nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
212 return res;
213 }
214 }
