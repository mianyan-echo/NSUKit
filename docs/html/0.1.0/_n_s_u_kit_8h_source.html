135 itf_ds_typesafe = std::is_base_of<I_BaseStreamUItf, DSItf_t>::value;
136 mw_cmd_typesafe = std::is_base_of<I_BaseRegMw, CmdMw_t>::value;
137 mw_stream_typesafe = std::is_base_of<I_BaseStreamMw, ChnlMw_t>::value;
138
139 itf_cs = new CSItf_t();
140 if (std::is_same<CSItf_t, CRItf_t>::value) {
141 itf_cr = itf_cs;
142 } else {
143 itf_cr = new CRItf_t();
144 }
145 itf_ds = new DSItf_t();
146 mw_cmd = new CmdMw_t(this);
147 mw_chnl = new ChnlMw_t(this);
148 }

180 template<class CSItf_t, class CRItf_t, class DSItf_t, class CmdMw_t, class ChnlMw_t>
181 nsukitStatus_t NSUSoc<CSItf_t, CRItf_t, DSItf_t, CmdMw_t, ChnlMw_t>::link_cmd(nsuInitParam_t *param) {
182 if (param == nullptr) {
183 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
184 }
185 METHOD_NEED_(check_typesafe);
186
187 // 初始化状态为成功
188 nsukitStatus_t status = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
189
190 if (!this->combined_cmd_itf()) {
191 auto* crInterface = dynamic_cast<I_BaseCmdUItf*>(itf_cr);
192 status |= crInterface->accept(param);
193 }
194 // 调用命令接口的 accept 函数，将结果合并到状态中
195 auto* csInterface = dynamic_cast<I_BaseCmdUItf*>(itf_cs);
196 status |= csInterface->accept(param);
197 // 调用命令中间件的 config 函数，将结果合并到状态中
198 auto* cmdMiddleware = dynamic_cast<I_BaseRegMw*>(mw_cmd);
199 status |= cmdMiddleware->config(param);
200
201 // 返回合并后的状态
202 return status;
203 }
