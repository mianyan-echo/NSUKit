45Mixin_NativeRegCmd::increment_write(nsuRegAddr_t addr, nsuVoidBuf_p value, nsuSize_t length, nsuSize_t reg_len) {
46 auto res = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
47 auto num = length / reg_len;
48 auto ptr = (nsuRegValue_t *) value;
49 for (int i=0;i<num;i++){
50 res |= cmd_itf_->write(addr+reg_len*i, *(ptr+i));
51 }
52 if ((length - num*reg_len) > 0) {
53 nsuRegValue_t last = 0;
54 memcpy(&last, ptr+num, length - num*reg_len);
55 res |= cmd_itf_->write(addr+reg_len*num, last);
56 }
57 return res;
58}

149nsukitStatus_t Mixin_VirtualRegCmd::_common_write(nsuRegAddr_t reg, nsuRegValue_t value) {
150 auto buf = (nsuCharBuf_p) malloc(sizeof(RegPack));
151 auto len = _fmt_reg_write(reg, value, buf);
152 auto res = cmd_itf_->send_bytes(buf, len);
153 free(buf);
154
155 char r_head[16];
156 RegPack pack;
157 res |= cmd_itf_->recv_bytes(16, r_head);
158 memcpy(&pack, r_head, 16);
159 res |= cmd_itf_->recv_bytes(pack.length-16, r_head);
160 if (*(nsuRegValue_t *)r_head!=0) res |= nsukitStatus_t::NSUKIT_STATUS_ITF_FAIL;
161 return res;
162}

165 if (buf == nullptr) return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
166 auto cmd_buf = (nsuCharBuf_p) malloc(sizeof(RegPack));
167 auto len = _fmt_reg_read(reg, cmd_buf);
168 auto res = cmd_itf_->send_bytes(cmd_buf, len);
169 free(cmd_buf);
170
171 char r_head[16];
172 RegPack pack;
173 res |= cmd_itf_->recv_bytes(16, r_head);
174 memcpy(&pack, r_head, 16);
175 res |= cmd_itf_->recv_bytes(pack.length-16, r_head);
176 if (*(nsuRegValue_t *)r_head!=0) res |= nsukitStatus_t::NSUKIT_STATUS_ITF_FAIL;
177 if (res == nsukitStatus_t::NSUKIT_STATUS_SUCCESS) *buf = *(nsuRegValue_t *)r_head;
178 return res;

217 RegPack head = {0x5F5F5F5F, 0x31001010, 0x00000000, padding_len+6*4, addr, padding_len};
218 auto buf = (nsuCharBuf_p ) malloc(sizeof(RegPack)+padding_len);

224 free(buf);

227 res |= cmd_itf_->recv_bytes(16, r_head);
