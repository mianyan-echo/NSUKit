102nsuMemory_p SimStreamUItf::alloc_buffer(nsuStreamLen_t length, nsuVoidBuf_p buf) {
103 bool self_malloc = false;
104 if (buf == nullptr) {
105 buf = malloc(length);
106 self_malloc = true;
107 }
108 auto memory = new DMAMemory{length, self_malloc, buf};
109 fdMap[memory] = memory;

113nsukitStatus_t SimStreamUItf::free_buffer(nsuMemory_p fd) {
114 if (fd == nullptr) {
115 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
116 }
117 auto memory = fdMap[fd];
118 fdMap.erase(fd);
119 if (memory->self_malloc) {
120 free(memory->buffer);
121 }
122 free(memory);
123 return nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
124}
