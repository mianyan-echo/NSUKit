45nsukitStatus_t parallel_execute(std::vector<nsukit::BaseKit *>& slave_list, const std::string& cmd) {
46 nsukitStatus_t res = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
47 unsigned long long num_parallel = slave_list.size();
48#ifndef HSYNC_EN_ASYNC
49 // thread库实现
50 std::thread threads[num_parallel];
51 for(int i=0; i < num_parallel; i++) {
52 const auto &elem = slave_list[i];
53 threads[i] = std::thread(&nsukit::BaseKit::execute, elem, cmd);
54// [&elem, cmd]() {elem->execute(cmd);}
55 }
56 for(int i=0; i < num_parallel; i++) {
57 threads[i].join();
58 }
59#else
60 // async库实现
61 std::future<nsukitStatus_t> futures[num_parallel];
62 for(int i=0; i<num_parallel; i++) {
63 const auto &elem = slave_list[i];
64 futures[i] = std::async(std::launch::async, &nsukit::BaseKit::execute, elem, cmd);
65 }
66 for(int i=0; i<num_parallel; i++) {
67 res |= futures[i].get();
68 }
69#endif
