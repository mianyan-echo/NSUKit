53template <typename T>
54class ThreadSafeQueue {
55public:
56 ThreadSafeQueue() {}
57
58 //
59 void Push(T *value) {
60 std::unique_lock<std::mutex> lock(mutex_);
61 queue_.push(value);
62 lock.unlock();
63 condition_.notify_one();
64 }
65
66 //
67 T *Pop() {
68 std::unique_lock<std::mutex> lock(mutex_);
69 condition_.wait(lock, [this] { return !queue_.empty(); });
70 auto value = (T *)queue_.front();
71 queue_.pop();
72 return value;
73 }
74
75private:
76 std::queue<T *> queue_;
77 std::mutex mutex_;
78 std::condition_variable condition_;
79};

100void upload_thread(nsukit::BaseKit *_kit, Deque *q, nsuSize_t block, nsuSize_t total, std::mutex *mu) {
101 std::unique_lock<std::mutex> *lock;
102 nsuMemory_p mem;
103 nsuSize_t current = 0;
104 while (true) {
105 mem = q->empty.Pop();
106 auto s = _kit->open_recv(0, mem, block, 0);
107 if (s != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
108 std::cout << "Establish CS and CR connections: " << std::endl;
109 std::cout << "Failed to enable data uplink " << nsukit::status2_string(s) << ", currently uplinked: " << current << std::endl;
110 break;
111 }
112 s = nsukitStatus_t::NSUKIT_STATUS_STREAM_RUNNING;
113 while (s==nsukitStatus_t::NSUKIT_STATUS_STREAM_RUNNING) {
114 s = _kit->wait_stream(mem, 1.);
115 }
116 current += block;
117 q->full.Push(mem);
118 if (current == total) break;
119 }
120 lock = new std::unique_lock<std::mutex>(*mu);
121 q->stopFlag = 1;
122 delete lock;
123}

134void write_file_thread(nsukit::BaseKit *_kit, Deque *q, nsuSize_t block, const std::string &path, std::mutex *mu) {
135 std::unique_lock<std::mutex> *lock;
136 nsuMemory_p mem;
137 std::ofstream outF;
138 outF.open(path, std::ofstream::binary);
139 int cnt = 0;
140 nsuSize_t speed_count = 0;
141 auto st = std::chrono::steady_clock::now();
142
143 while (true) {
144 lock = new std::unique_lock<std::mutex>(*mu);
145 if (q->stopFlag == 1) {
146 break;
147 }
148 delete lock;
149 mem = q->full.Pop();
150 outF.write((char *)_kit->get_buffer(mem, block), block);
151 q->empty.Push(mem);
152// std::this_thread::sleep_for(std::chrono::seconds(1));
153 speed_count += block;
154
155 if (cnt % 20 == 0) {
156 auto count = std::chrono::steady_clock::now() - st;
157 std::cout << std::flush << '\r' << "current write file speed: " << speed_count*1000./(count.count()) << "MB/s" << std::endl;
158 speed_count = 0;
159 st = std::chrono::steady_clock::now();
160 }
161 cnt++;
162 }
163 std::cout << std::endl;
164 outF.close();
165}

168int main(int argc, char *argv[]) {
169 unsigned int ds_block = 1024*1024;
170 Deque q;
171 std::mutex mu;
172 SocType kit{};
173
174 if (argc != 4) {
175 std::cout << "Unsupported parameter passing method" << std::endl;
176 // DataUpload 127.0.0.1 104857600 ./da_data_1.dat
177 std::cout << argv[0] << " {IP} {totalBytes} {filePath}" << std::endl;
178 return 1;
179 }
180 nsuSize_t total_len = std::atoi(argv[2]);
181 if (total_len % ds_block != 0) {
182 std::cout << "The total length of upstream data total_len "
183 << total_len << " Bytes should be "
184 << ds_block << "Integer multiple of Bytes" << std::endl;
185 return 1;
186 }
187
188 nsuInitParam_t param;
189 param.cmd_ip = argv[1];
190 param.cmd_board = 0;
191 param.stream_board = 0;
192#ifdef STREAM_WITH_TCP
193 auto ip = std::string(argv[1]);
194 // 192.168.1.161 => 6001
195 std::string port_str{};
196 port_str += ip[ip.length()-2];
197 port_str += "00";
198 port_str += ip[ip.length()-1];
199 int port = std::atoi(port_str.data());
200 param.stream_ip = ip;
201 param.stream_tcp_port = port;
202 param.stream_tcp_block = 4 * 1024 * 1024;
203#endif
204
205 auto res = kit.link_cmd(&param);
206 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
207 std::cout << "Establish CS and CR connections: " << nsukit::status2_string(res) << std::endl;
208 }
209
210 res = kit.link_stream(&param);
211 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
212 std::cout << "Establish a DS connection: " << nsukit::status2_string(res) << std::endl;
213 }
214
215 std::cout << "SocLink Successful!!! now alloc buffer" << std::endl;
216 for (int i=0; i<10; i++) {
217 nsuMemory_p mem = kit.alloc_buffer(ds_block);
218 q.empty.Push(mem);
219 }
220
221 res = kit.execute("RFConfig");
222 std::cout << "Soc Init Successful!!!" << std::endl;
223
224 kit.set_param("TotalTriggerPoints", 128); // 128k sample points
225 kit.set_param("PreTriggerPoints", 512); // pre sample 512 points
226 kit.set_param("CH0TriggerEnable", 1); // enable CH0 Trigger
227 kit.set_param("CH0TriggerLevel", 4096); // config CH0 Trigger level
228 res = kit.execute("ADTriggerConfig");
229 std::cout << "ADTrigger Config Successful!!!" << std::endl;
230
231#ifdef STREAM_WITH_TCP
232 kit.set_param("ADC数据输出方式", 1);
233 std::cout << "Use TCP get data stream" << std::endl;
234#else
235 kit.set_param("ADC数据输出方式", 0);
236 std::cout << "Use PCIE get data stream" << std::endl;
237#endif
238 // Notify FPGA to start data acquisition.
239 res = kit.execute("SocStart");
240 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
241 std::cout << "SocStart: " << nsukit::status2_string(res) << std::endl;
242 }
243
244 // Start the upstream and storage threads.
245 std::thread up_trd(upload_thread, &kit, &q, ds_block, total_len, &mu);
246 std::thread write_trd(write_file_thread, &kit, &q, ds_block, argv[3], &mu);
247 up_trd.join();
248 write_trd.join();
249
250 // Notify FPGA to stop data acquisition.
251 res = kit.execute("SocStop");
252 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
253 std::cout << "SocStop: " << nsukit::status2_string(res) << std::endl;
254 }
255
256 std::cout << "Data upload completed" << std::endl;
257
258 return 0;
259}
