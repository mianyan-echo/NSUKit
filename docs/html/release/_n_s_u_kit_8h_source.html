295 template<class CSItf_t, class CRItf_t, class DSItf_t, class CmdMw_t, class ChnlMw_t>
296 NSUSoc<CSItf_t, CRItf_t, DSItf_t, CmdMw_t, ChnlMw_t>::NSUSoc() {
297 itf_cs_typesafe = std::is_base_of<I_BaseCmdUItf, CSItf_t>::value;
298 itf_cr_typesafe = std::is_base_of<I_BaseCmdUItf, CRItf_t>::value;
299 itf_ds_typesafe = std::is_base_of<I_BaseStreamUItf, DSItf_t>::value;
300 mw_cmd_typesafe = std::is_base_of<I_BaseRegMw, CmdMw_t>::value;
301 mw_stream_typesafe = std::is_base_of<I_BaseStreamMw, ChnlMw_t>::value;
302
303 itf_cs = new CSItf_t();
304 if (std::is_same<CSItf_t, CRItf_t>::value) {
305 itf_cr = (CRItf_t *)itf_cs;
306 } else {
307 itf_cr = new CRItf_t();
308 }
309 itf_ds = new DSItf_t();
310 mw_cmd = new CmdMw_t(itf_cs);
311 mw_chnl = new ChnlMw_t(itf_ds);
312 }

327 template<class CSItf_t, class CRItf_t, class DSItf_t, class CmdMw_t, class ChnlMw_t>
328 nsukitStatus_t NSUSoc<CSItf_t, CRItf_t, DSItf_t, CmdMw_t, ChnlMw_t>::link_cmd(nsuInitParam_t *param) {
329 if (param == nullptr) {
330 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
331 }
332 METHOD_NEED_(check_typesafe);
333
334 //
335 nsukitStatus_t status = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
336
337 if (!this->combined_cmd_itf()) {
338 auto* crInterface = dynamic_cast<I_BaseCmdUItf*>(itf_cr);
339 status |= crInterface->accept(param);
340 }
341 //
342 auto* csInterface = dynamic_cast<I_BaseCmdUItf*>(itf_cs);
343 status |= csInterface->accept(param);
344 //
345 auto* cmdMiddleware = dynamic_cast<I_BaseRegMw*>(mw_cmd);
346 status |= cmdMiddleware->config(param);
347
348 if (status == nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
349 cmd_linked = true;
350 }
351
352 //
353 return status;
354 }

358 nsukitStatus_t NSUSoc<CSItf_t, CRItf_t, DSItf_t, CmdMw_t, ChnlMw_t>::unlink_cmd() {
359 METHOD_NEED_(check_typesafe);
360 //
361 nsukitStatus_t status = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
362
363 if (!this->combined_cmd_itf()) {
364 auto* crInterface = dynamic_cast<I_BaseCmdUItf*>(itf_cr);
365 status |= crInterface->close();
366 }
367 status |= dynamic_cast<I_BaseCmdUItf *>(itf_cs)->close();
368 if (status == nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
369 cmd_linked = false;
370 }
371 return status;
372 }
