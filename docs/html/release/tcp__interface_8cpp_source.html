133nsuMemory_p TCPStreamUItf::alloc_buffer(nsuStreamLen_t length, nsuVoidBuf_p buf) {
134 auto mem = new Memory{};
135 if (buf == nullptr) {
136 mem->memory = (nsuMemory_p) malloc(length);
137 mem->self_alloc = true;
138 } else {
139 mem->memory = (nsuMemory_p) buf;
140 mem->self_alloc = false;
141 }
142
143 memset(mem->memory, length, 1);
144 mem->idx = memory_index;
145 mem->mem_size = length;
146 mem->finish_event.setEvent();
147
148 memory_dict[memory_index] = mem;
149 memory_index++;
150 return mem;
151}

173nsukitStatus_t
174TCPStreamUItf::open_recv(nsuChnlNum_t chnl, nsuMemory_p fd, nsuStreamLen_t length, nsuStreamLen_t offset) {
175 auto res = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
176 auto mem = (Memory *) fd;
177 { // set memory param
178 std::unique_lock<std::mutex> lock(mem->mtx);
179 if (mem->mem_size < length + offset) {
180 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
181 }
182 mem->need_size = length;
183 mem->using_size = 0;
184 mem->error_msg = "";
185 }
186
187 {
188 std::unique_lock<std::mutex> lock(opLock);
189 if ((!sockGen.tcpClientConnected) && sockGen.AcceptClient()==-1) return nsukitStatus_t::NSUKIT_STATUS_ACCEPT_FAIL;
190 }
191
192 auto thread = std::thread(_recv, &sockGen, &stop_event, mem, block_size, length, offset);
193 if (upload_thread.joinable()) upload_thread.join();
194 upload_thread = std::move(thread);
195 return res;
196}

199void TCPStreamUItf::_recv(SocketGenerator *sock, ThreadSafeEvent *stop_event, Memory *mem, uint32_t block,
200 nsuStreamLen_t length, nsuStreamLen_t offset) {
201 auto buffer = (nsuCharBuf_p) (mem->memory) + offset;
202 mem->finish_event.resetEvent();
203 stop_event->resetEvent();
204 int stLen = 0;
205 int totalLen = 0;
206 int cur_count;
207
208 while (length) {
209 if (stop_event->isSet()) {
210 mem->finish_event.setEvent();
211 break;
212 }
213 cur_count = length - totalLen > block ? block : length - totalLen;
214 stLen = sock->RecvData(sock->tcpClient, buffer + totalLen, cur_count);
215 if (stLen == 0) {
216 // server connect interrupt
217 std::unique_lock<std::mutex> lock(mem->mtx);
218 mem->error_msg += "connect interrupt error";
219 break;
220 } else if (stLen > 0) {
221 std::unique_lock<std::mutex> lock(mem->mtx);
222 totalLen += stLen;
223 mem->using_size = totalLen;
224 if (totalLen >= length) {
225 break;
226 }
227 }
228 }
229 mem->finish_event.setEvent();
230 stop_event->setEvent();
231}

234nsukitStatus_t TCPStreamUItf::wait_stream(nsuMemory_p fd, float timeout) {
235 auto mem = (Memory *) fd;
236 if (memory_dict.count(mem->idx) == 0) {
237 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
238 }
239 mem->finish_event.waitForEvent(static_cast<int>(timeout));
240
241 std::unique_lock<std::mutex> lock(mem->mtx);
242 if (mem->using_size < mem->need_size) {
243 return nsukitStatus_t::NSUKIT_STATUS_STREAM_RUNNING;
244 } else if (!mem->error_msg.empty()) {
245 return nsukitStatus_t::NSUKIT_STATUS_STREAM_FAIL;
246 } else {
247 return nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
248 }

262TCPStreamUItf::stream_recv(nsuChnlNum_t chnl, nsuMemory_p fd, nsuStreamLen_t length, nsuStreamLen_t offset,
263 bool (*stop_event)(), float timeout, int flag) {
264 auto res = open_recv(chnl, fd, length, offset);
265 if (stop_event == nullptr) stop_event = []() {return false;};
266 res = nsukitStatus_t::NSUKIT_STATUS_STREAM_RUNNING;
267 while (!stop_event() && res!=nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
268 res = wait_stream(fd, 1.);
269 }
270 return res;
271}
