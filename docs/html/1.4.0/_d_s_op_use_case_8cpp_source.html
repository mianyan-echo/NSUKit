22int main(int argc, char *argv[]) {
23 const int ds_chnl = 0;
24 std::ofstream outf;
25 nsukit::NSUSoc<nsukit::TCPCmdUItf, nsukit::PCIECmdUItf, nsukit::PCIEStreamUItf> kit{};
26 nsuInitParam_t param;
27
28 std::cout << "当前示例" << argv[0]
29 << "为展示基于PCIE的数据上行的接口调用示例，功能为从板卡0的数传通道0上行指定数量的数据并写入文件" << std::endl;
30 if (argc != 4) {
31 std::cout << "不受支持的传参方法" << std::endl;
32 std::cout << argv[0] << " {IP} {totalBytes} {filePath}" << std::endl;
33 return 1;
34 }
35 nsuSize_t total_len = std::atoi(argv[2]);
36 if (total_len % 4 != 0) {
37 std::cout << "上行数据总长度total_len " << total_len << "Bytes应为 " << 4 << "Bytes的整倍数" << std::endl;
38 return 1;
39 }
40
41 // 连接板卡
42 param.cmd_ip = argv[1];
43 param.cmd_board = 0;
44 param.stream_board = 0;
45 auto res = kit.link_cmd(&param);
46 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
47 std::cout << "建立CS、CR连接：" << nsukit::status2_string(res) << std::endl;
48 }
49 res = kit.link_stream(&param);
50 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
51 std::cout << "建立DS连接：" << nsukit::status2_string(res) << std::endl;
52 }
53
54 // 准备DS连接要用的内存
55 nsuMemory_p mem = kit.alloc_buffer(total_len); // 申请内存
56 auto ds_buf = (char *)kit.get_buffer(mem, total_len); // 获取内存首指针
57
58 // 通知FPGA开始采集
59 res = kit.execute("系统开启");
60 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
61 std::cout << "系统开启：" << nsukit::status2_string(res) << std::endl;
62 }
63
64 // 上行一个颗粒度的数据
65 res = kit.open_recv(ds_chnl, mem, total_len, 0);
66 if (res == nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
67 std::cout << "成功开启DS交互" << std::endl;
68 auto ds_state = nsukitStatus_t::NSUKIT_STATUS_STREAM_RUNNING;
69 while (ds_state != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
70 ds_state = kit.wait_stream(mem, 1000.);
71// std::cout << "wait_stream：" << nsukit::status2_string(ds_state) << std::endl;
72 }
73 }
74
75 std::cout << *(uint32_t *)ds_buf << std::endl;
76
77 // 通知FPGA停止采集
78 res = kit.execute("系统停止");
79 if (res != nsukitStatus_t::NSUKIT_STATUS_SUCCESS) {
80 std::cout << "系统停止：" << nsukit::status2_string(res) << std::endl;
81 }
82
83 // 存盘
84 outf.open(argv[3], std::ofstream::binary);
85 outf.write(ds_buf, total_len);
86 outf.close();
87
88 kit.free_buffer(mem); // 释放内存
89 return 0;
90}
