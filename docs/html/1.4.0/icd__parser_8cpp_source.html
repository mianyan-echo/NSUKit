13inline static void *CheckFeedback(char *data) {
14 StandardCmdFeedback feedback;
15 memcpy(&feedback, data, sizeof(feedback));
16 if (feedback.packHead != 0xCFCFCFCF) {
17 std::cout << "Receive instruction header is misplaced" << std::endl;
18 return nullptr;
19 }
20 if (feedback.execSucceed != 0) {
21 std::cout << "The command was sent successfully but failed to execute" << std::endl;
22 return nullptr;
23 }
24 std::cout << "Command sent successfully" << std::endl;
25 return nullptr;
26}

29bool JsonWrapper::InitICD(std::string_view path) {
30 // Get the input stream of icd.json
31 std::ifstream _in(path.data(), std::ios::binary);
32 if (!_in.is_open()) {
33 std::cout << "can not find icd file: " << path << std::endl;
34 throw std::runtime_error("can not find icd file");
35 }
36
37 // Parse json file
38 if (ICDReader.parse(_in, icdRoot)) {
39 icdCommands = &icdRoot["command"];
40 if (icdRoot.isMember("param")) {
41 icdParams = &icdRoot["param"];
42 }
43 } else {
44 std::cout << "file: " << path << "Wrong format!" << std::endl;
45 throw std::runtime_error("icd file format error");
46 }
47 _in.close();
48
49 return true;
50}

189nsukitStatus_t ICDRegMw::execute(nsuCSName_t cname) {
190 auto res = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
191 CommandPack *pCommandPack;
192 try {
193 pCommandPack = FmtCommand(cname);
194 } catch (...) {
195 return nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
196 }
197
198 res |= cs_itf->send_bytes(pCommandPack->bytes, pCommandPack->length);
199
200 if (pCommandPack->recvLen > 0) {
201 char *result = new char[pCommandPack->recvLen]();
202 res |= cs_itf->recv_bytes(pCommandPack->recvLen, result);
203 if (checkFeedback) CheckFeedback(result);
204 res |= EnableParam(cname, result);
205 delete[] result;
206 }
207 DeleteCommandPack(pCommandPack);
208 return res;
209}

218CommandPack *JsonWrapper::FmtRegister(const Json::Value &reg, const Json::Value &value) {
219 if (value.isString() && StringStartWith(value.asString(), "0x")) {
220 // 对于 "0x5F5F5F5F"
221 auto _value = (uint32_t) std::stoll(value.asString(), nullptr, 16);
222 return FmtRegister(reg, _value);
223 } else if (value.isString() && StringStartWith(value.asString(), "0b")) {
224 // 对于 "0b10101011"
225 auto _value = std::stoll(value.asString().substr(2), nullptr, 2);
226 return FmtRegister(reg, _value);
227 } else if (value.isString() && reg[0].asString() == ICD_File) {
228 // 对于 file文件类型
229 return FmtFile(value.asString());
230 } else if (value.isString() && reg[0].asString() == ICD_FileLength) {
231 // 对于文件大小类型
232 return FmtFile(value.asString(), true);
233 } else if (value.isNumeric()) {
234 // 对于全体数字
235 std::string value_type = reg[0].asString();
236 if (value_type == ICD_Float) {
237 auto _value = value.asFloat();
238 return FmtRegister(reg, _value);
239 } else if (value_type == ICD_Double) {
240 auto _value = value.asDouble();
241 return FmtRegister(reg, _value);
242 } else if (value_type == ICD_Uint8) {
243 auto _value = (uint8_t) value.asUInt();
244 return FmtRegister(reg, _value);
245 } else if (value_type == ICD_Int8) {
246 auto _value = (int8_t) value.asInt();
247 return FmtRegister(reg, _value);
248 } else if (value_type == ICD_Uint16) {
249 auto _value = (uint16_t) value.asUInt();
250 return FmtRegister(reg, _value);
251 } else if (value_type == ICD_Int16) {
252 auto _value = (int16_t) value.asInt();
253 return FmtRegister(reg, _value);
254 } else if (value_type == ICD_Uint32) {
255 auto _value = (uint32_t) value.asUInt();
256 return FmtRegister(reg, _value);
257 } else if (value_type == ICD_Int32) {
258 auto _value = (int32_t) value.asInt();
259 return FmtRegister(reg, _value);
260 } else if (value_type == ICD_FileLength) {
261 auto _value = (uint32_t) value.asInt();
262 return FmtRegister(reg, _value);
263 } else {
264 std::cout << "Unrecognized value type " << value.toStyledString() << std::endl;
265 throw "Unrecognized register value type";
266 }
267 } else {
268 std::cout << "Unrecognized value type " << value.toStyledString() << std::endl;
269 throw "Unrecognized register value type";
270 }
271}

299CommandPack *JsonWrapper::FmtFile(const std::string &file_path, bool need_size) {
300 CommandPack *fileData = nullptr;
301 std::ifstream _in(file_path, std::ios::in | std::ios::binary | std::ios::ate);
302 if (_in.is_open()) {
303 fileData = new CommandPack;
304 if (need_size) {
305 fileData->length = TypeSizeMap[ICD_Uint32];
306 fileData->bytes = new char[fileData->length + 1]();
307 uint32_t fileLength = (uint32_t) _in.tellg();
308 memcpy(fileData->bytes, &fileLength, fileData->length);
309 } else {
310 fileData->length = _in.tellg();
311 fileData->bytes = new char[fileData->length + 1]();
312 _in.seekg(0, std::ios::beg);
313 _in.read(fileData->bytes, fileData->length);
314 }
315 } else {
316 std::cout << "The file path " << file_path << " is wrong" << std::endl;
317 }
318 return fileData;
319}

328CommandPack *ICDRegMw::FmtCommand(const std::string &cmd_name, const std::string &file_path) {
329 Json::Value command;
330 std::vector<CommandPack *> registerList;
331 CommandPack *regTemp;
332 nsuSize_t recv_len = 0;
333 auto resCommandPack = new CommandPack;
334 uint32_t commandLength = 0;
335
336 if (!icd->icdCommands->isMember(cmd_name)) {
337 std::cout << "Not find command: " << cmd_name << std::endl;
338 throw std::runtime_error("The instruction name does not exist in ICD");
339 }
340
341 // 生成要发送的指令
342 command = (*(icd->icdCommands))[cmd_name]["send"];
343 for (auto &reg: command) {
344 if (reg.isString() && reg.asString() == ICD_ParamFlag_File) {
345 regTemp = icd->FmtFile(file_path);
346 } else if (reg.isString() && reg.asString() == ICD_ParamFlag_FileLength) {
347 regTemp = icd->FmtFile(file_path, true);
348 } else if (reg.isString() && icd->icdParams->isMember(reg.asString())) {
349 reg = (*icd->icdParams)[reg.asString()];
350 regTemp = icd->FmtRegister(reg, reg[1]);
351 } else if (reg.isArray()) {
352 regTemp = icd->FmtRegister(reg, reg[1]);
353 } else {
354 throw std::runtime_error("Sorry, send to registers is unsupported");
355 }
356 assert(regTemp->bytes != nullptr);
357 registerList.emplace_back(regTemp);
358 commandLength += regTemp->length;
359 }
360
361 // 计算要接收的指令长度
362 command = (*(icd->icdCommands))[cmd_name]["recv"];
363 for (auto &reg: command) {
364 if (reg.isString() && reg.asString() == ICD_ParamFlag_File) {
365 throw std::runtime_error("Sorry, recv from registers is unsupported");
366 } else if (reg.isString() && reg.asString() == ICD_ParamFlag_FileLength) {
367 throw std::runtime_error("Sorry, recv from registers is unsupported");
368 } else if (reg.isString() && icd->icdParams->isMember(reg.asString())) {
369 reg = (*(icd->icdParams))[reg.asString()];
370 recv_len += TypeSizeMap[reg[0].asString()];
371 } else if (reg.isArray()) {
372 recv_len += TypeSizeMap[reg[0].asString()];
373 } else {
374 throw std::runtime_error("Sorry, recv from registers is unsupported");
375 }
376 }
377
378 // 将获取到的二进制寄存器列表拼接为一个CommandPack
379 resCommandPack->bytes = new char[commandLength + 1];
380 resCommandPack->length = commandLength;
381 resCommandPack->recvLen = recv_len;
382 char *flag = resCommandPack->bytes;
383 for (auto &reg: registerList) {
384 memcpy(flag, reg->bytes, reg->length);
385 flag += reg->length;
386 DeleteCommandPack(reg);
387 }
388 // 修改指令长度
389 memcpy(resCommandPack->bytes + 12, &commandLength, 4);
390
391 return resCommandPack;
392}

394nsukitStatus_t ICDRegMw::EnableParam(nsuCSName_t cname, nsuCharBuf_p buf) {
395 Json::Value command;
396 auto res = nsukitStatus_t::NSUKIT_STATUS_SUCCESS;
397
398 if (!icd->icdCommands->isMember(cname)) {
399 res |= nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
400 throw std::runtime_error("指令名称在icd中不存在");
401 }
402
403 // 遍历recv中的所有参数
404 command = (*(icd->icdCommands))[cname]["recv"];
405 auto ptr = (nsuCharBuf_p )buf;
406 for (auto &reg: command) {
407 if (reg.isString() && reg.asString() == ICD_ParamFlag_File) {
408 res |= nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
409 } else if (reg.isString() && reg.asString() == ICD_ParamFlag_FileLength) {
410 res |= nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
411 } else if (reg.isString() && icd->icdParams->isMember(reg.asString())) {
412 auto _pack = (*(icd->icdParams))[reg.asString()];
413 auto _width = TypeSizeMap[_pack[0].asString()];
414 res |= set_param_by_tname(reg.asString(), _pack[0].asString(), ptr);
415 ptr += _width;
416 } else if (reg.isArray()) {
417 ptr += TypeSizeMap[reg[0].asString()];
418 } else {
419 res |= nsukitStatus_t::NSUKIT_STATUS_INVALID_VALUE;
420 }
421 }
422 return res;
423}
